64 constexpr bool check_boundary_correctness() const {
65 if (m_boundaries.size() < 2) {
66 return true;

68 namespace rs = std::ranges;
69 namespace vs = std::views;
70 auto b_n_plus_1_view = m_boundaries | vs::drop(1);
71 auto b_n_view = m_boundaries | vs::take(m_boundaries.size() - 1);
72 return rs::all_of(
73 vs::zip(b_n_plus_1_view, b_n_view),
74 [](auto const& boundaries) {
75 try {
76 auto const& [b_n_plus_1, b_n] = boundaries;

129template<EuclideanDomain T>
130Homology<T> homology(ChainComplex<T> const& chain_complex) {
131 namespace rs = std::ranges;
132 namespace vs = std::views;
133 auto const& boundaries = chain_complex.boundaries();
134 Homology<T> homology;
135 homology.betti_numbers.resize(boundaries.size());
136 homology.torsion.resize(boundaries.size());
137 std::size_t prev_smith_units_count = 0;
138 std::vector<T> prev_smith_diagonal_without_units {};
139 for (std::size_t k = boundaries.size(); k > 0; --k) {
140 auto const n = k - 1;
141 auto const& boundary = boundaries[n];
142 auto [smith, rank] = smith_form(boundary);
143 auto nullity = boundary.ncols() - rank;
144 auto smith_diagonal_without_units = rs::to<std::vector>(
145 vs::iota(0u, rank)
146 | vs::transform([&smith](std::size_t n) { return smith[n, n]; })
147 | vs::drop_while([](T const& x) {
148 return x.euclidean_function() == 1;
149 })
150 );
151 auto smith_units = rank - smith_diagonal_without_units.size();
152 homology.betti_numbers[n] = nullity - prev_smith_units_count
153 - prev_smith_diagonal_without_units.size();
154 homology.torsion[n] = std::move(prev_smith_diagonal_without_units);
155 prev_smith_units_count = smith_units;
156 prev_smith_diagonal_without_units =
157 std::move(smith_diagonal_without_units);
158 }
159 return homology;
160}

164template<Field T>
165Homology<T> homology(ChainComplex<T> const& chain_complex) {
166 auto const& boundaries = chain_complex.boundaries();
167 Homology<T> homology;
168 homology.betti_numbers.resize(boundaries.size());
169 homology.torsion.resize(boundaries.size());
170 std::size_t prev_rank = 0;
171 for (std::size_t k = boundaries.size(); k > 0; --k) {
172 auto const n = k - 1;
173 auto const& boundary = boundaries[n];
174 auto [_, rank] = row_echelon_form(boundary);
175 auto nullity = boundary.ncols() - rank;
176 homology.betti_numbers[n] = nullity - prev_rank;
177 prev_rank = rank;
178 }
179 return homology;
