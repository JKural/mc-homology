20constexpr std::size_t row_echelon_form(std::in_place_t, Matrix<T>& matrix) {
21 std::size_t i = 0;
22 for (std::size_t j = 0; j < matrix.ncols(); ++j) {
23 auto maybe_i =
24 detail::first_nonzero_submatrix_column_coefficient(matrix, i, j);
25 if (!maybe_i) {
26 continue;
27 }
28 if (*maybe_i != i) {
29 detail::submatrix_swap_rows(matrix, i, *maybe_i, j);
30 }
31 for (auto k = i + 1; k < matrix.nrows(); ++k) {
32 auto mult = -matrix[k, j] / matrix[i, j];
33 detail::submatrix_add_row(matrix, mult, i, k, j);
34 }
35 ++i;
36 }
37 return i;
38}

78template<EuclideanDomain T>
79constexpr std::size_t smith_form(std::in_place_t, Matrix<T>& matrix) {
80 std::size_t k = 0;
81 auto move_min_to_corner = [&] {
82 auto min_element =
83 detail::minimal_nonzero_submatrix_element(matrix, k, k);
84 if (matrix[min_element.first, min_element.second] == T::zero()) {
85 return;
86 }
87 if (min_element.first != k) {
88 detail::submatrix_swap_rows(matrix, k, min_element.first, k);
89 }
90 if (min_element.second != k) {
91 detail::submatrix_swap_cols(matrix, k, min_element.second, k);
92 }
93 };
94 for (; k < std::min(matrix.nrows(), matrix.ncols()); ++k) {
95 bool col_all_zeros = false;
96 while (!col_all_zeros) {
97 move_min_to_corner();
98 if (matrix[k, k] == T::zero()) {
99 return k;
100 }
101 col_all_zeros = true;
102 for (std::size_t i = k + 1; i < matrix.nrows(); ++i) {
103 auto [q, r] = divide(matrix[i, k], matrix[k, k]);
104 detail::submatrix_add_row(matrix, -q, k, i, k);
105 if (r != T::zero()) {
106 col_all_zeros = false;
107 }
108 }
109 }
110 bool row_all_zeros = false;
111 while (!row_all_zeros) {
112 move_min_to_corner();
113 if (matrix[k, k] == T::zero()) {
114 return k;
115 }
116 row_all_zeros = true;
117 for (std::size_t j = k + 1; j < matrix.ncols(); ++j) {
118 auto [q, r] = divide(matrix[k, j], matrix[k, k]);
119 detail::submatrix_add_col(matrix, -q, k, j, k);
120 if (r != T::zero()) {
121 row_all_zeros = false;
122 }
123 }
124 }
125 }
126 if constexpr (std::totally_ordered<T>) {
127 for (std::size_t i = 0; i < k; ++i) {
128 if (matrix[i, i] < T::zero()) {
129 matrix[i, i] = -matrix[i, i];
130 }
131 }
132 }
133 return k;
134}
