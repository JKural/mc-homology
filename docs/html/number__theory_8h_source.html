101constexpr ExtendedGCDResult extended_gcd(int a, int b) noexcept {
102 a = std::abs(a);
103 b = std::abs(b);
104 int x1 = 1;
105 int x2 = 0;
106 int y1 = 0;
107 int y2 = 1;

111 std::tie(x1, x2) = std::pair {x2, x1 - q * x2};
112 std::tie(y1, y2) = std::pair {y2, y1 - q * y2};
113 }
114 return {.g = a, .x = x1, .y = y1};
