341template<Ring T>
342constexpr Matrix<T> operator*(Matrix<T> const& lhs, Matrix<T> const& rhs) {
343 using Matrix = Matrix<T>;
344 using size_type = Matrix::size_type;
345 if (lhs.ncols() != rhs.nrows()) {
346 throw std::domain_error(
347 "The number of columns of lhs is different "
348 "than the number of rows of rhs"
349 );
350 }
351 auto const inner_dim = lhs.ncols();
352 Matrix product = Matrix::zero(lhs.nrows(), rhs.ncols());
353 for (size_type i = 0; i < product.nrows(); ++i) {
354 for (size_type j = 0; j < product.ncols(); ++j) {
355 for (size_type k = 0; k < inner_dim; ++k) {
356 product[i, j] += lhs[i, k] * rhs[k, j];
357 }
358 }
359 }
360 return product;
361}
